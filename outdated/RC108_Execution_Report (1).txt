    syndrome_probs = np.ones(n_syndromes) / n_syndromes
    logical_probs = []
    measured_gens = set()

    for t in range(n_ticks):
        gen_idx = schedule_func(t, H.shape[0])
        measured_gens.add(gen_idx)

        # Apply projector: zero out syndromes with bit gen_idx = 1
