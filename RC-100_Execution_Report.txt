"""
RC-100 Complete Reproduction Script
Tests whether the S-involution acts as a non-trivial logical Clifford gate

Requires: numpy

import numpy as np
from itertools import combinations

# ============================================================================
# STEP 1: Build G24 (cyclic construction)
# ============================================================================
g = np.array([1, 0, 1, 0, 1, 1, 1, 0, 0, 0, 1, 1] + [0]*11, dtype=int)

G23 = np.zeros((12, 23), dtype=int)
for i in range(12):
    for j in range(23):
        G23[i, j] = g[(j - i) % 23]

G24 = np.zeros((12, 24), dtype=int)
for i in range(12):
    parity = np.sum(G23[i]) % 2
    G24[i] = np.hstack([G23[i], [parity]])

# Verify weight distribution
from collections import defaultdict
weights = defaultdict(int)
for bits in range(2**12):
    coeffs = np.array([(bits >> i) & 1 for i in range(12)], dtype=int)
    cw = (coeffs @ G24) % 2
    weights[int(np.sum(cw))] += 1

assert weights == {0: 1, 8: 759, 12: 2576, 16: 759, 24: 1}
assert np.all((G24 @ G24.T) % 2 == 0)

# ============================================================================
# STEP 2: Build S-involution
# ============================================================================
def mobius_involution(x):
    if x == 0: return 23
    elif x == 23: return 0
    else: return (-pow(x, -1, 23)) % 23

S_perm = np.zeros((24, 24), dtype=int)
for i in range(24):
    S_perm[mobius_involution(i), i] = 1

assert np.array_equal(S_perm @ S_perm % 2, np.eye(24, dtype=int))

# Verify S preserves code
code_words = set()
for bits in range(2**12):
    coeffs = np.array([(bits >> i) & 1 for i in range(12)], dtype=int)
    cw = tuple((coeffs @ G24) % 2)
    code_words.add(cw)

for cw in code_words:
    cw_arr = np.array(cw)
    mapped = (S_perm @ cw_arr) % 2
    assert tuple(mapped) in code_words

# ============================================================================
# STEP 3: Construct logical operators
# ============================================================================
C_basis = G24.copy()

# Find coset representatives for F2^24 / C
full_basis = list(C_basis)
coset_basis = []
for i in range(24):
    e_i = np.zeros(24, dtype=int)
    e_i[i] = 1
    test = np.array(full_basis + [e_i])
    if np.linalg.matrix_rank(test) > len(full_basis):
        full_basis.append(e_i)
        coset_basis.append(e_i)
    if len(coset_basis) == 12:
        break

# Build overlap matrix M[i,j] = u_i · g_j
M = np.zeros((12, 12), dtype=int)
for i in range(12):
    for j in range(12):
        M[i, j] = (coset_basis[i] @ C_basis[j]) % 2

# Invert M over F2
def gf2_inv(matrix):
    A = matrix.copy() % 2
    n = A.shape[0]
    aug = np.hstack([A, np.eye(n, dtype=int)])
    for col in range(n):
        pivot = None
        for row in range(col, n):
            if aug[row, col] == 1:
                pivot = row
                break
        if pivot is None:
            raise ValueError("Singular")
        aug[[col, pivot]] = aug[[pivot, col]]
        for row in range(n):
            if row != col and aug[row, col] == 1:
                aug[row] = (aug[row] + aug[col]) % 2
    return aug[:, n:]

M_inv = gf2_inv(M)

# Build logical operators
X_logical = []
Z_logical = []
for i in range(12):
    u_i = coset_basis[i]
    X_logical.append((u_i.copy(), u_i.copy()))

    coeffs = M_inv[:, i]
    d_i = np.zeros(24, dtype=int)
    for k in range(12):
        if coeffs[k]:
            d_i = (d_i + C_basis[k]) % 2

    Z_logical.append((u_i.copy(), (u_i + d_i) % 2))

# ============================================================================
# STEP 4: Compute S action on logical operators
# ============================================================================
def act_S(a, b):
    return (S_perm @ a) % 2, (S_perm @ b) % 2

def gf2_solve(A, b):
    A = A.copy() % 2
    b = b.copy() % 2
    m, n = A.shape
    aug = np.hstack([A, b.reshape(-1, 1)])
    rank = 0
    pivot_cols = []
    for col in range(n):
        pivot = None
        for row in range(rank, m):
            if aug[row, col] == 1:
                pivot = row
                break
        if pivot is None:
            continue
        aug[[rank, pivot]] = aug[[pivot, rank]]
        for row in range(m):
            if row != rank and aug[row, col] == 1:
                aug[row] = (aug[row] + aug[rank]) % 2
        pivot_cols.append(col)
        rank += 1
    if any(aug[row, n] == 1 for row in range(rank, m)):
        return False, None
    x = np.zeros(n, dtype=int)
    for idx in range(len(pivot_cols) - 1, -1, -1):
        col = pivot_cols[idx]
        r = idx
        x[col] = aug[r, n]
        for c in range(col + 1, n):
            if aug[r, c] == 1:
                x[col] = (x[col] + x[c]) % 2
    return True, x

# Build matrices
U_mat = np.array(coset_basis).T  # 24 x 12
D_mat = np.zeros((24, 12), dtype=int)
for j in range(12):
    d_j = (Z_logical[j][0] + Z_logical[j][1]) % 2
    D_mat[:, j] = d_j

def decompose(a, b):
    sum_ab = (a + b) % 2
    in_C, _ = gf2_solve(C_basis.T, sum_ab)
    if not in_C:
        return None
    exists_w, w = gf2_solve(D_mat, sum_ab)
    if not exists_w:
        return None
    aug_matrix = np.hstack([U_mat, C_basis.T])
    exists_aug, coeffs_aug = gf2_solve(aug_matrix, a)
    if not exists_aug:
        return None
    vw = coeffs_aug[:12]
    v = (vw + w) % 2
    return v, w

# Compute S action
A = np.zeros((12, 12), dtype=int)  # X->X
B = np.zeros((12, 12), dtype=int)  # X->Z
C_mat = np.zeros((12, 12), dtype=int)  # Z->X
D_mat_out = np.zeros((12, 12), dtype=int)  # Z->Z

for i in range(12):
    sx = act_S(X_logical[i][0], X_logical[i][1])
    result = decompose(sx[0], sx[1])
    if result:
        A[i], B[i] = result

    sz = act_S(Z_logical[i][0], Z_logical[i][1])
    result = decompose(sz[0], sz[1])
    if result:

def gf2_det(matrix):
    A = matrix.copy() % 2
    n = A.shape[0]
    det = 1
    for i in range(n):
        pivot = None
